# build
cd /Users/serjik/work/Dev/LearnKubernetes/sampleApp
docker build -t sgdpro/sample-app:in-mem .
docker build -t sgdpro/sample-app:in-file .

# run the in memory sample
docker run -d -p 3080:3080 sgdpro/sample-app:in-mem
docker run -d -p 3080:3080 sgdpro/sample-app:in-file

docker volume create books-storage
docker run --rm --name appfile -d -p 3080:3080 -v books-storage:/usr/src/data sgdpro/sample-app:in-file

# post books
curl -X GET http://localhost:3080/books
curl -X POST http://localhost:3080/books -H "Content-Type: application/json" -H "Conent-type: application/json" -d '{"name": "book3", "author": "smith"}'
